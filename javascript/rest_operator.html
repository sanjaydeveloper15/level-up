ES6 - Rest Operator

Rest operator means passing multiple value in comma seprated from in function and they can get as Object ...args

function multiplies(){
//for in way
for (var i in arguments) {
document.write(arguments[i]);
}
document.write('
');
for (var i = 0; i < arguments.length; i++) {
document.write(arguments[i]);
}
document.write('
');
let res = 1;
for(let i in arguments){
res *= arguments[i];
}
document.write(`Final result via for in: ${res}`);
} multiplies(1,2,3,4);
