ES6 - Spread Operator

Spread operator in mostly using for Array while calling we need to give ... for function for get individual values

function multiplies(){
//for in way
for (var i in arguments) {
document.write(arguments[i]);
}
document.write('
');
for (var i = 0; i < arguments.length; i++) {
document.write(arguments[i]);
}
document.write('
');
let res = 1;
for(let i in arguments){
res *= arguments[i];
}
document.write(`Final result via for in: ${res}`);
}
var arr = [1,2,3,4,5];
multiplies(...arr);
